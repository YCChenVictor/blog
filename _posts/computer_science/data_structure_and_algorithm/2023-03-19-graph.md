```javascript
class Graph {
  constructor() {
    this.nodes = new Map();
  }

  addNode(node) {
    this.nodes.set(node, []);
  }

  addEdge(nodeA, nodeB) {
    if (!this.nodes.has(nodeA)) {
      this.addNode(nodeA);
    }
    if (!this.nodes.has(nodeB)) {
      this.addNode(nodeB);
    }
    this.nodes.get(nodeA).push(nodeB);
    this.nodes.get(nodeB).push(nodeA);
  }

  getNeighbors(node) {
    return this.nodes.get(node);
  }
}

// Create a new graph
const graph = new Graph();

// Add nodes and edges to the graph
graph.addEdge("A", "B");
graph.addEdge("B", "C");
graph.addEdge("C", "D");
graph.addEdge("D", "E");
graph.addEdge("E", "F");
graph.addEdge("F", "G");
graph.addEdge("G", "A");

// Get the neighbors of node "C"
const neighborsOfC = graph.getNeighbors("C");
console.log(neighborsOfC); // Output: ["B", "D"]
```
