### key, value

```javascript
const object1 = {
  a: 'somestring',
  b: 42,
  c: false
};

// key
console.log(Object.keys(object1));

// value
console.log(Object.values(object1));
```

### join

```javascript
const elements = ['Fire', 'Air', 'Water'];

console.log(elements.join());
// expected output: "Fire,Air,Water"

console.log(elements.join(''));
// expected output: "FireAirWater"

console.log(elements.join('-'));
// expected output: "Fire-Air-Water"
```
